\textbf{Head-Linear Reductions (Small-Step)}
\vskip1cm
$$
\withenv{x:B,\;\Gamma;\:\Delta}{\trans{x}{}{B}}
\ruleno{B-Var}
$$

$$
\ctrans{\withenv{\Gamma;\:\Delta}{\trans{x}{}{x}}}
       {x\ne dom(\Gamma)}
\ruleno{F-Var}
$$
