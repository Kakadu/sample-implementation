\vskip1cm
\textbf{Call by Name (Small-Step)}
\vskip1cm

Substitution:

$$
x [z\gets B] = \left\{
                 \begin{array}{rcl}
                    B&,&z=x\\
                    x&,&z\ne x
                 \end{array}
               \right.
$$
$$
X [x\gets B] = X
$$
$$
(M\:N)[x\gets A]=(M[x\gets A])(N[x\gets A])
$$
$$
(\lambda z.B)[x\gets A]=\left\{
                   \begin{array}{rcl}
                     \lambda z.B&,&z=x\\
                     \lambda z.(B[x\gets A)&,&z\ne x
                   \end{array}
                 \right.
$$
\vskip 5mm
Reduction rules:

$$
\trans{(\lambda x.A)\:B}{}{A[x\gets B]}
\ruleno{Red}
$$

$$
\crule{\trans{M}{}{M^\prime}}
      {\trans{M\:N}{}{M^\prime\:N}}
      {M\ne \lambda x.A}
\ruleno{App}
$$
